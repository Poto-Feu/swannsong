ROOM menu
    TITLE "SwannSong Adventure"
    DESC menu_desc
    ATLAUNCH
        DISPLAY ALL

        IF g_firstlaunch NOT EXISTS
            SET g_firstlaunch = true
        END
    END

    CHOICES
        c1
            TEXT menu_c1

            IF g_firstlaunch = true
                CUTSCENE welcome
                g_firstlaunch = false
            END

            GO house
            GET 3 Coin
        END
        c2
            TEXT menu_c2

            CUTSCENE about
        END
    END
END ROOM

ROOM house
    TITLE house_title
    DESC house_desc
    ATLAUNCH
        DISPLAY ALL
    END
    CHOICES
        c1
            TEXT house_c1
            UNFINISHED
        END
        c2
            TEXT house_c2
            UNFINISHED
        END
    END
END ROOM
