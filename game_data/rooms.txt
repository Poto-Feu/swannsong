ROOM menu
    TITLE "SwannSong Adventure"
    DESC menu_desc
    ATLAUNCH
        DISPLAY TITLE
        DISPLAY DESC
        DISPLAY CHOICES

        IF firstlaunch NOT EXISTS
            SET firstlaunch = false
        END
    END

    CHOICES
        c1
            TEXT menu_c1

            firstlaunch = true

            IF firstlaunch = true
                CUTSCENE welcome
            END

            GO house
            GET 3 Coin
        END
        c2
            TEXT menu_c2

            CUTSCENE about
        END
    END
END ROOM

ROOM house
    TITLE house_title
    DESC house_desc
    ATLAUNCH
        DISPLAY TITLE
        DISPLAY DESC
        DISPLAY CHOICES
    END
    CHOICES
        c1
            TEXT house_c1
            UNFINISHED
        END
        c2
            TEXT house_c2
            UNFINISHED
        END
    END
END ROOM
